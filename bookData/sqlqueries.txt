-- library.author definition

CREATE TABLE `author` (
  `auth_id` bigint NOT NULL AUTO_INCREMENT,
  `f_name` varchar(100) NOT NULL,
  `l_name` varchar(100) NOT NULL,
  `last_updated` timestamp NULL DEFAULT NULL,
  `valid_flag` char(1) NOT NULL,
  PRIMARY KEY (`auth_id`)
) ENGINE=InnoDB AUTO_INCREMENT=51 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;

-- library.book definition

CREATE TABLE `book` (
  `book_id` bigint NOT NULL AUTO_INCREMENT,
  `year_of_publishment` year NOT NULL,
  `book_name` varchar(500) DEFAULT NULL,
  `last_updated` timestamp NULL DEFAULT NULL,
  `availability` char(1) NOT NULL,
  `valid_flag` char(1) NOT NULL,
  `auth_id` bigint NOT NULL,
  `loc_id` bigint NOT NULL,
  `pages` int NOT NULL,
  `author_id` int DEFAULT NULL,
  PRIMARY KEY (`book_id`),
  KEY `book_FK` (`auth_id`),
  KEY `book_FK_2` (`loc_id`),
  CONSTRAINT `book_FK` FOREIGN KEY (`auth_id`) REFERENCES `author` (`auth_id`) ON DELETE CASCADE ON UPDATE CASCADE,
  CONSTRAINT `book_FK_2` FOREIGN KEY (`loc_id`) REFERENCES `book_loc_genre` (`loc_id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB AUTO_INCREMENT=890 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
-- library.book_loc_genre definition

CREATE TABLE `book_loc_genre` (
  `genre` varchar(50) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `last_updated` timestamp NULL DEFAULT NULL,
  `loc_id` bigint NOT NULL AUTO_INCREMENT,
  PRIMARY KEY (`loc_id`)
) ENGINE=InnoDB AUTO_INCREMENT=24 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
